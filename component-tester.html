Buttons

Button Variants
teaUI.createButton({ text: 'Primary', variant: 'primary' }) teaUI.createButton({ text: 'Secondary', variant: 'secondary' }) teaUI.createButton({ text: 'Outline', variant: 'outline' }) teaUI.createButton({ text: 'Text', variant: 'text' })
Button Sizes
teaUI.createButton({ text: 'Small', size: 'small' }) teaUI.createButton({ text: 'Medium', size: 'medium' }) teaUI.createButton({ text: 'Large', size: 'large' })
Full Width Button
teaUI.createButton({ text: 'Full Width Button', fullWidth: true })
Disabled Button
teaUI.createButton({ text: 'Disabled Button', disabled: true })

Toggles

Basic Toggle
teaUI.createToggle({ checked: false })
Toggle with Labels
teaUI.createToggle({ checked: false, labelLeft: 'Western', labelRight: 'Gongfu' })
Custom Color Toggle
teaUI.createToggle({ checked: true, color: '#7B9070' // Green Tea color })

Tabs

Default Tabs
teaUI.createTabs({ tabs: [ { label: 'Details', content: 'Details tab content' }, { label: 'Brewing', content: 'Brewing tab content' }, { label: 'Notes', content: 'Notes tab content' } ] })
Pills Tabs
teaUI.createTabs({ tabs: [ { label: 'Details', content: 'Details tab content' }, { label: 'Brewing', content: 'Brewing tab content' }, { label: 'Notes', content: 'Notes tab content' } ], variant: 'pills' })

Cards

Basic Card
teaUI.createCard({ title: 'Card Title', content: 'This is the card content' })
Card with Footer
teaUI.createCard({ title: 'Card with Footer', content: 'This card has a footer section', footer: 'Card Footer' })
Card Elevation Levels
teaUI.createCard({ title: 'Low Elevation', content: 'Low shadow', elevation: 'low' }) teaUI.createCard({ title: 'Medium Elevation', content: 'Medium shadow', elevation: 'medium' }) teaUI.createCard({ title: 'High Elevation', content: 'High shadow', elevation: 'high' })

Badges

Badge Variants
teaUI.createBadge({ text: 'Default', variant: 'default' }) teaUI.createBadge({ text: 'Primary', variant: 'primary' }) teaUI.createBadge({ text: 'Success', variant: 'success' }) teaUI.createBadge({ text: 'Warning', variant: 'warning' }) teaUI.createBadge({ text: 'Error', variant: 'error' })
Pill Badges
teaUI.createBadge({ text: 'Default', pill: true }) teaUI.createBadge({ text: 'Primary', variant: 'primary', pill: true }) teaUI.createBadge({ text: 'Success', variant: 'success', pill: true })
Outline Badges
teaUI.createBadge({ text: 'Default', outline: true }) teaUI.createBadge({ text: 'Primary', variant: 'primary', outline: true }) teaUI.createBadge({ text: 'Success', variant: 'success', outline: true })

Chips

Basic Chips
teaUI.createChip({ text: 'Basic Chip' })
Closable Chips
teaUI.createChip({ text: 'Closable Chip', closable: true })
Custom Color Chips
teaUI.createChip({ text: 'Green Tea', backgroundColor: '#7B9070', color: 'white' }) teaUI.createChip({ text: 'Black Tea', backgroundColor: '#A56256', color: 'white' })

Progress Bars

Basic Progress Bar
teaUI.createProgressBar({ value: 50 })
Progress Bar with Percentage
teaUI.createProgressBar({ value: 75, showPercentage: true })
Striped Progress Bar
teaUI.createProgressBar({ value: 60, striped: true })
Animated Striped Progress Bar
teaUI.createProgressBar({ value: 80, striped: true, animated: true })
Custom Color Progress Bar
teaUI.createProgressBar({ value: 40, color: '#7B9070' // Green tea color })
